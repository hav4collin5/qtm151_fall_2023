def fn_mse(y, x, b0, b1):
    e2 = (y - b0 - b1 * x)**2
    return e2.mean()

y = fifa_data['Potential']
x = fifa_data['Age']

mse_params1 = fn_mse(y, x, 80, 0.4)

# Write your own code

list_status = [0, 1]
list_prob = [0.5, 0.5]
n = len(fifa_data)

np.random.seed(42)
fifa_data['status'] = np.random.choice(list_status, size=n, p=list_prob)

data0 = fifa_data.query('status == 0')
data1 = fifa_data.query('status == 1')
